```python
# 샘플 데이터의 각 문서를 단어장과 비교하며 training될 matrix를 생성합니다.

trainMat = []
for postinDoc in listOPosts:

```


```python
# trainNB0 함수를 사용하여 세 개의 값을 반환받습니다.

p0V, p1V, pAb = naivebayes.trainNB0(trainMat, listClasses)
```
